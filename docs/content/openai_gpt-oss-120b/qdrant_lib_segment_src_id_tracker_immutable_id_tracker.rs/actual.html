fn mmap_size(len: usize) -> usize {
    let item_width = std::mem::size_of::();
    let chunks = len.div_ceil(item_width);
    chunks * item_width
}

fn bitmap_mmap_size(number_of_elements: usize) -> usize {
    const BITS_TO_BYTES: usize = 8; // Convert bits to bytes
    mmap_size::(number_of_elements.div_ceil(BITS_TO_BYTES))
}

    /// Example:
    ///   0.4.0 -> 0.4.1 = true

    ///   0.4.0 -> 0.5.0 = false
    ///   0.4.0 -> 0.5.1 = fla
    pub fn can_upgrade_storage(stored: &Version, app: &Version) -> bool {
        if stored.major != app.major {
            return false;
        }
        if stored.minor != app.minor {
            return false;
        }
        if stored.patch + 1 < app.patch {
            return false;
        }
        true
    }
